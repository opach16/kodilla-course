SELECT L.NAME AS LIST, I.SUMMARY AS ISSUE FROM ISSUES I
JOIN ISSUESLISTS L ON I.ISSUESLIST_ID = L.ID
ORDER BY L.ID;

SELECT U.FIRSTNAME, U.LASTNAME, I.SUMMARY AS ISSUE FROM ISSUES I
JOIN USERS U ON USER_ID_ASSIGNEDTO = U.ID
ORDER BY U.FIRSTNAME, U.LASTNAME;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS NUMBER_OF_ISSUES FROM ISSUES I
JOIN USERS U ON USER_ID_ASSIGNEDTO = U.ID
GROUP BY I.USER_ID_ASSIGNEDTO
ORDER BY U.FIRSTNAME, U.LASTNAME;
